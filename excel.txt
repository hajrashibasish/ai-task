# 3) Quick sanity on versions actually installed
python - <<'PY'
import importlib.metadata as m
for name in ("h11","httpcore","httpx","uvicorn","starlette","fastapi","langchain","openai"):
    try:
        print(f"{name}: {m.version(name)}")
    except m.PackageNotFoundError:
        pass
PY
