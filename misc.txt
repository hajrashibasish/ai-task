def test_save_metadata_xml_calls_feed_method_fixed(load_module):
    # Patch the static method on the class, not on the instance
    with patch('WGPT.metadata_processing.MetadataProcessing.save_metadata_feed') as mock_feed:
        mock_db = MagicMock()
        file_info = {"feed_source": "xml", "type": "text"}

        load_module.save_metadata(
            url="url",
            language="en",
            doc_id="id",
            title="title",
            file_info=file_info,
            db=mock_db,
            etl_schema="etl"
        )

        mock_feed.assert_called_once()
