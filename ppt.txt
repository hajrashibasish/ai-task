def test_process_ppt_to_markdown(load_module):
    mock_ppt = MagicMock()
    mock_instance = MagicMock()
    # âœ… Return two values as expected by the main function
    mock_instance.convert.return_value = ("slides.md", 123)
    mock_ppt.return_value = mock_instance

    return_values = {}
    file_path = "/ppt/slides.pptx"
    output_dir = "/output"

    with patch.object(load_module, "Ppt2Markdown", mock_ppt):
        result = load_module.process_ppt_to_markdown(file_path, output_dir, return_values)

    assert isinstance(result, dict)
    assert result.get("status") == "Successful"
    assert result.get("document_length") == 123
    assert result.get("md_file_location") == "slides.md"
